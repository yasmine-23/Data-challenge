library(dplyr)

## Warning: le package 'knitr' a été compilé avec la version R 4.4.2
# Créer une table récapitulative
covid_summary <- covid_data %>%
  group_by(Country_Region) %>%
  summarise(
    cas_confirmes = sum(Confirmed, na.rm = TRUE),
    cas_deces = sum(Deaths, na.rm = TRUE),
    cas_retablis = sum(Recovered, na.rm = TRUE),
    cas_actifs = sum(Active, na.rm = TRUE),
    avg_case_fatality_ratio = mean(Case_Fatality_Ratio, na.rm = TRUE)
  ) %>%
  arrange(desc(cas_confirmes))  # Trier par nombre de cas confirmés

#Calcul des pourcentages

# Totaux mondiaux
world_totals <- covid_summary %>%
  summarise(
    cas_confirmes = sum(cas_confirmes, na.rm = TRUE),
    cas_deces = sum( cas_deces, na.rm = TRUE),
    cas_retablis = sum(cas_retablis, na.rm = TRUE),
    cas_actifs = sum(cas_actifs, na.rm = TRUE)
  )

# Totaux pour l'Union européenne
eu_totals <- eu_summary %>%
  summarise(
    cas_confirmes = sum(cas_confirmes, na.rm = TRUE),
    cas_deces = sum(cas_deces, na.rm = TRUE),
    cas_retablis = sum(cas_retablis, na.rm = TRUE),
    cas_actifs = sum(cas_actifs, na.rm = TRUE)
  )

# Fusionner les deux jeux de données 
comparison <- bind_rows(
  world_totals %>% mutate(Region = "Monde"),
  eu_totals %>% mutate(Region = "Union européenne")
) %>%
  pivot_longer(
    cols = -Region,
    names_to = "Variable",
    values_to = "Value"
  ) %>%
  group_by(Variable) %>%
  mutate(Percentage = Value / sum(Value) * 100)

# Barplot comparatif pour toutes les variables
ggplot(comparison, aes(x = Variable, y = Percentage, fill = Region)) +
  geom_bar(stat = "identity", position = "dodge") +
  labs(
    title = "Comparaison des indicateurs COVID-19 entre le monde et l'Union européenne",
    x = "Variable",
    y = "Pourcentage",
    fill = "Région"
  ) +
  theme_minimal() +
  scale_fill_manual(values = c("Monde" = "orange", "Union européenne" = "blue"))

server <- function(input, output, session) {
  # Génération de la carte interactive
  output$map <- renderLeaflet({
    # Palette de couleurs dynamique basée sur l'indicateur sélectionné
    pal <- colorNumeric("YlOrRd", domain = covid_data_clean[[input$indicator]])
    
    # Création de la carte
    leaflet(data = covid_data_clean) %>%
      addTiles() %>%  # Ajouter une couche de fond
      addCircles(
        lng = ~Long_, lat = ~Lat,  # Coordonnées des pays
        weight = 1,
        radius = ~get(input$indicator) / max(get(input$indicator), na.rm = TRUE) * 500000,
        color = ~pal(get(input$indicator)),
        popup = ~paste(
          "<strong>Pays :</strong>", Country_Region, "<br>",
          "<strong>", input$indicator, " :</strong>", get(input$indicator)
        ),
        fillOpacity = 0.7
      ) %>%
      addLegend(
        pal = pal, values = ~get(input$indicator),
        title = paste("Nombre de", tolower(input$indicator)),
        position = "bottomright"
      ) %>%
      setView(lng = 10, lat = 50, zoom = 4)  # Centrer la vue sur l'Europe
  })
}

library(shiny)

# Liste des pays de l'Union Européenne
ue_countries <- c(
  "Austria", "Belgium", "Bulgaria", "Croatia", "Cyprus", "Czech Republic",
  "Denmark", "Estonia", "Finland", "France", "Germany", "Greece", 
  "Hungary", "Ireland", "Italy", "Latvia", "Lithuania", "Luxembourg", 
  "Malta", "Netherlands", "Poland", "Portugal", "Romania", "Slovakia", 
  "Slovenia", "Spain", "Sweden"
)

# Préparer les données pour l'évolution temporelle
covid_data_time <- covid_data %>%
  filter(Country_Region %in% ue_countries) %>%
  drop_na(Last_Update, Confirmed, Deaths, Active) %>%
  mutate(Last_Update = as.Date(Last_Update)) %>%
  group_by(Country_Region, Last_Update) %>%
  summarise(
    Confirmed = sum(Confirmed, na.rm = TRUE),
    Deaths = sum(Deaths, na.rm = TRUE),
    Active = sum(Active, na.rm = TRUE),
    .groups = "drop"
  )

# Interface utilisateur (UI)
ui <- fluidPage(
  titlePanel("Évolution temporelle des cas COVID-19 dans l'Union Européenne"),
  
  sidebarLayout(
    sidebarPanel(
      selectInput(
        "country", 
        "Choisir un pays :", 
        choices = unique(covid_data_time$Country_Region),
        selected = "France"
      ),
      selectInput(
        "indicator", 
        "Choisir un indicateur :", 
        choices = c(
          "Cas confirmés" = "Confirmed",
          "Décès" = "Deaths",
          "Cas actifs" = "Active"
        ),
        selected = "Confirmed"
      ),
      helpText("Sélectionnez un pays et un indicateur pour afficher son évolution dans le temps.")
    ),
    
    mainPanel(
      plotlyOutput("time_plot", height = "600px")
    )
  )
)

# Serveur
server <- function(input, output, session) {
  output$time_plot <- renderPlotly({
    country_data <- covid_data_time %>%
      filter(Country_Region == input$country)
    
    plot_ly(
      data = country_data,
      x = ~Last_Update,
      y = ~get(input$indicator),
      type = 'scatter',
      mode = 'lines+markers',
      line = list(color = 'blue'),
      marker = list(size = 6, color = 'red')
    ) %>%
      layout(
        title = paste("Évolution des", input$indicator, "pour", input$country),
        xaxis = list(title = "Date"),
        yaxis = list(title = input$indicator)
      )
  })
}

# Lancer l'application Shiny
shinyApp(ui = ui, server = server)
